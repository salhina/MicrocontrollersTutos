#include<htc.h>
#define  _XTAL_FREQ  4000000
__CONFIG(WDTDIS&UNPROTECT&MCLRDIS);
unsigned int i;

//****fonction de temporisation : front descendant sur enable
void enable(void)
{
RC1=1;
for (i=0;i<4000;i++)
{}
RC1=0;
}

//****la fonction principale main
void main()
{
   //******configuration des entrées sorties
TRISC=0b00000000;
TRISD=0b00000000;
   RC0=0;    // RS=0 selectioner le registre de commande pour LCD
//fonction clear display
PORTD=0b00000001;
enable(); // validation de la commande clear display

//fonction set
PORTD=0b00111100;
enable();

//disply on/off
PORTD=0b00001111;
enable();

//Entry mode set
PORTD=0b00000110;
enable();

//fonction DD ram
PORTD=0b11000101;
enable();

   RC0=1;    // RS=1  écrire dans le  LCD
PORTD='N';      enable();
PORTD='A';      enable();
PORTD='B';      enable();
PORTD='I';      enable();
PORTD='L';      enable();

while (1) // pour que l’affichage reste pour long temps
   {}
}
